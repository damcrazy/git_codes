Scaffold with safe area
import 'package:flutter/cupertino.dart';
import 'package:flutter/material.dart';
void main() {
runApp(MaterialApp(
home: Scaffold(
appBar: AppBar(
title: Text("Scaffold usage WITH SAFE AREA"),
),
body: Container(
child: SafeArea(
child: Center(
child: Text(
'This is scaffold area',style: TextStyle(
fontWeight: FontWeight.bold,
),
),
),
),
),
),
),);
}
